const THRESH={'Rebel':[3,5,7],'Star Guardian':[2,4,6],'K/DA':[2,4,6],'Academy':[2,4,6],'Bruiser':[2,4],'Bastion':[2,4,6],'Gunner':[2,4,6],'Slayer':[2,4,6],'Caster':[2,4,6],'True Damage':[2,3],'Technogenius':[1,2]};
const CC={1:'#9aa4af',2:'#1dc93d',3:'#4a9eff',4:'#c267ff',5:'#f0b429'};
const W=70,H=81,hw=35,hh=40.5,rg=H*.75,PAD=8,NS='http://www.w3.org/2000/svg';
const ROWS=[7,6,7,6];

function makeHexes(){
  const h=[];
  ROWS.forEach((cnt,r)=>{
    const off=r%2?hw:0, cy=PAD+hh+r*rg;
    for(let c=0;c<cnt;c++) h.push({id:`${r}-${c}`,cx:PAD+hw+off+c*W,cy});
  });
  return h;
}
const hexes=makeHexes();

let pBoard={},eBoard={};
hexes.forEach(h=>{pBoard[h.id]=null;eBoard[h.id]=null;});
let selHex=null,selType=null,showNames=true,showTraits=true,showEnemy=false,unitsOpen=false,activeTarget='player';
let hist=[],fut=[];

function snap(){hist.push({p:{...pBoard},e:{...eBoard}});if(hist.length>40)hist.shift();fut=[];}
function hexPts(cx,cy){return[[cx,cy-hh],[cx+hw,cy-hh/2],[cx+hw,cy+hh/2],[cx,cy+hh],[cx-hw,cy+hh/2],[cx-hw,cy-hh/2]].map(p=>p.join(',')).join(' ');}
function trunc(s,n){return s.length>n?s.slice(0,n-1)+'â€¦':s;}
function getBoard(t){return t==='enemy'?eBoard:pBoard;}

function renderSVG(el,board,type){
  el.innerHTML='';
  hexes.forEach(h=>{
    const un=board[h.id],unit=un?UNITS.find(u=>u.name===un):null;
    const isSel=selType===type&&selHex===h.id;

    g.setAttribute('class',`hg ${type}${un?' occ':''}${isSel?' sel':''}`);
    const poly=document.createElementNS(NS,'polygon');
    poly.setAttribute('points',hexPts(h.cx,h.cy));
    g.appendChild(poly);
    if(unit&&showNames){
      const col=CC[unit.cost];
      const mk=(y,fs,txt,fw)=>{const t=document.createElementNS(NS,'text');t.setAttribute('x',h.cx);t.setAttribute('y',y);t.setAttribute('text-anchor','middle');t.setAttribute('font-size',fs);t.setAttribute('fill',col);if(fw)t.setAttribute('font-weight',fw);t.textContent=txt;return t;};
      g.appendChild(mk(h.cy-3,'8.5',trunc(un,11),'600'));
      g.appendChild(mk(h.cy+9,'7.5','$'+unit.cost,null));
    } else if(!unit){
      const plus=document.createElementNS(NS,'text');
      plus.setAttribute('x',h.cx);plus.setAttribute('y',h.cy+5);plus.setAttribute('text-anchor','middle');
      plus.setAttribute('font-size','16');plus.setAttribute('fill',type==='enemy'?'#251515':'#1a1a32');
      plus.textContent='+';g.appendChild(plus);
    }
    g.addEventListener('click',()=>onHex(h.id,type));
    el.appendChild(g);
  });
}

function onHex(id,type){
  const board=getBoard(type);
  if(selHex&&selType===type){
    if(selHex===id){selHex=null;selType=null;render();return;}
    snap();
    const tmp=board[selHex];board[selHex]=board[id];board[id]=tmp;
    selHex=null;selType=null;
  } else if(selHex&&selType!==type){
    selHex=null;selType=null;
  } else {
    if(board[id]){selHex=id;selType=type;}
  }
  render();
}

function render(){
  renderSVG(document.getElementById('player-board'),pBoard,'player');
  if(showEnemy) renderSVG(document.getElementById('enemy-board'),eBoard,'enemy');
  updateStats();
  renderTraits();
  if(unitsOpen) renderUnitsPanel();
  const hint=document.getElementById('move-hint');
  hint.style.display=selType==='player'?'inline':'none';
}

function updateStats(){
  const pu=Object.values(pBoard).filter(Boolean);
  document.getElementById('player-count').textContent=pu.length;
  document.getElementById('player-cost').textContent='$'+pu.reduce((s,n)=>{const u=UNITS.find(x=>x.name===n);return s+(u?u.cost:0);},0);
  if(showEnemy){
    const eu=Object.values(eBoard).filter(Boolean);
    document.getElementById('enemy-count').textContent=eu.length;
    document.getElementById('enemy-cost').textContent='$'+eu.reduce((s,n)=>{const u=UNITS.find(x=>x.name===n);return s+(u?u.cost:0);},0);
  }
}

function renderTraits(){
  const ts=document.getElementById('traits-section');
  if(!showTraits){ts.style.display='none';return;}
  const counts={};
  Object.values(pBoard).filter(Boolean).forEach(n=>{const u=UNITS.find(x=>x.name===n);if(u)u.traits.forEach(t=>{counts[t]=(counts[t]||0)+1;});});
  const entries=Object.entries(counts).sort((a,b)=>b[1]-a[1]);
  if(!entries.length){ts.style.display='none';return;}
  ts.style.display='block';
  const list=document.getElementById('traits-list');
  list.innerHTML='';
  entries.forEach(([trait,cnt])=>{
    const th=THRESH[trait]||[],active=th.some(t=>cnt>=t),next=th.find(t=>cnt<t);
    const b=document.createElement('div');
    b.style.cssText=`padding:3px 9px;border-radius:4px;font-size:10px;font-weight:500;font-family:var(--font-sans);border:1px solid ${active?'#f0b429':'#252545'};background:${active?'rgba(240,180,41,.09)':'#0f0f22'};color:${active?'#f0b429':'#555'};`;
    b.textContent=trait+' '+cnt+(next?' â†’'+next:'');
    list.appendChild(b);
  });
}

function renderUnitsPanel(){
  const targetRow=document.getElementById('units-target-row');
  targetRow.style.display=showEnemy?'flex':'none';
  ['player','enemy'].forEach(t=>{
    const btn=document.getElementById('tgt-'+t);
    btn.classList.toggle('active',activeTarget===t);
  });
  const grid=document.getElementById('units-grid');
  grid.innerHTML='';
  const board=getBoard(activeTarget);
  const placed=new Set(Object.values(board).filter(Boolean));
  [5,4,3,2,1].forEach(cost=>{
    const units=UNITS.filter(u=>u.cost===cost);
    const col=CC[cost];
    const hdr=document.createElement('div');
    hdr.style.cssText=`color:${col};font-size:10px;font-weight:600;margin:7px 0 4px;font-family:var(--font-sans);`;
    hdr.textContent='$'+cost+' Â·';
    grid.appendChild(hdr);
    const row=document.createElement('div');
    row.style.cssText='display:flex;flex-wrap:wrap;gap:4px;';
    units.forEach(u=>{
      const on=placed.has(u.name);
      const btn=document.createElement('button');
      btn.className='upill'+(on?' on'+(activeTarget==='enemy'?' enemy':''):'');
      btn.innerHTML=`<span style="color:${col};font-weight:600;">${u.name}</span><span style="color:#444;font-size:9px;">${u.traits.join(', ')}</span>`;
      btn.addEventListener('click',()=>{
        const b=getBoard(activeTarget);
        const p=new Set(Object.values(b).filter(Boolean));
        snap();
        if(p.has(u.name)){
          const hid=hexes.find(h=>b[h.id]===u.name);
          if(hid) b[hid.id]=null;
        } else {
          const empty=hexes.find(h=>!b[h.id]);
          if(empty) b[empty.id]=u.name;
        }
        render();
      });
      row.appendChild(btn);
    });
    grid.appendChild(row);
  });
}
